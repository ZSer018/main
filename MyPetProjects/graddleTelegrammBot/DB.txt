#drop database Mimimicure;
#create database Mimimicure;
use Mimimicure;

drop table if exists portfolio;
drop table if exists service_data;
drop table if exists users;


create table users
(
	tg_user_id int not null primary Key,
	user_name VARCHAR(64) not null, 
	phone VARCHAR(14) not null, 
	tg_user VARCHAR(60) not null, 
	serv_count VARCHAR(5)
)CHARACTER SET utf8 COLLATE utf8_unicode_ci;


create table service_data
(
    id int primary key NOT NULL AUTO_INCREMENT,
	tg_user_id int not null,
	serv_type VARCHAR(30) not null, 
	reg_date VARCHAR(100) not null, 
	price int not null,
	FOREIGN KEY (tg_user_id) REFERENCES users (tg_user_id)
)CHARACTER SET utf8 COLLATE utf8_unicode_ci;


create table portfolio
(
    id int primary key NOT NULL AUTO_INCREMENT,
	img_category VARCHAR(30) not null, 
	img_link VARCHAR(500) not null
)CHARACTER SET utf8 COLLATE utf8_unicode_ci;
